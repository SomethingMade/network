rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /public/vynix.apk {
      allow read: if true;
      allow write: if false; // only you upload via console/CLI
    }

    // Group photos & videos — uploaded by openGroupChatView's media/caption flow.
    // Storage rules can't read Realtime Database, so we can't verify group
    // membership or "only owner can post" here — that's enforced in the app
    // (canPostInGroup) and should also be enforced in your RTDB rules for
    // group_threads/{groupId}/messages. This rule just requires sign-in.
    match /groups/{groupId}/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && request.resource.size < 100 * 1024 * 1024 // 100MB cap
                   && request.resource.contentType.matches('image/.*|video/.*');
    }

    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
